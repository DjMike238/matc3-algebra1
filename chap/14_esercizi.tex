\begin{multicols}{2}

\begin{enumeratea}

 \item $\dfrac{x}{y}+\dfrac{y}{y+1}=\dfrac{x+y^{2}}{y}$ \tab\boxV\quad\boxF
\end{enumeratea}
\end{multicols}
